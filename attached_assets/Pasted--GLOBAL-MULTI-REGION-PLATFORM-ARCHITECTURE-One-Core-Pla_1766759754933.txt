### Region Detection Logic
```javascript
// Auto-detect user region (multi-source)
async function detectUserRegion() {
  const detectionSources = [
    
    // 1. User's saved preference (highest priority)
    localStorage.getItem('userRegion'),
    
    // 2. IP-based geolocation (MaxMind, IP2Location)
    await fetch('https://api.ipgeolocation.io/ipgeo?apiKey=XXX'),
    
    // 3. Browser locale/language
    navigator.language, // 'en-US', 'hi-IN', etc.
    
    // 4. GPS (if permission granted)
    navigator.geolocation.getCurrentPosition(),
    
    // 5. URL subdomain/path
    window.location.hostname.split('.')[0], // 'in' from 'in.agriconnect.com'
    
    // 6. Accept-Language header (server-side)
    req.headers['accept-language'],
    
    // 7. Currency from payment method (if logged in)
    user.preferredCurrency,
    
    // 8. Default fallback
    'global'
  ];
  
  // Merge and prioritize
  return resolveBestMatch(detectionSources);
}
```

```javascript
async function searchProducts(query, userRegion) {
  
  // Step 1: Determine which database collection to query
  const collection = `products_${userRegion}`; // products_IN
  
  // Step 2: Apply region-specific filters
  const filters = {
    region: userRegion,
    status: 'active',
    $text: { $search: query }, // "tomatoes"
    location: {
      $near: {
        $geometry: userLocation.coordinates,
        $maxDistance: 50000 // 50km radius
      }
    }
  };
  
  // Step 3: Get user's preferred language
  const language = user.preferredLanguage; // 'hi' for Hindi
  
  // Step 4: Execute query
  const results = await db.collection(collection).find(filters);
  
  // Step 5: Return localized results
  return results.map(product => ({
    ...product,
    name: product.name[language] || product.name['en'], // Fallback to English
    price: formatPrice(product.price, user.preferredCurrency)
  }));
```
